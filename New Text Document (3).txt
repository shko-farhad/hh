#include <IRremote.h>

#define IR_PIN 2
#define LED_PIN 13

bool ledState = false;

void setup() {
  pinMode(LED_PIN, OUTPUT);
  Serial.begin(9600);

  IrReceiver.begin(IR_PIN, ENABLE_LED_FEEDBACK); // Start receiver
}

void loop() {
  if (IrReceiver.decode()) {

    Serial.println(IrReceiver.decodedIRData.decodedRawData, HEX);

    // Replace this with your button code
    if (IrReceiver.decodedIRData.decodedRawData == 0xEF10BF00) {
      ledState = !ledState;
      digitalWrite(LED_PIN, ledState);
    }

    IrReceiver.resume(); // Ready for next signal
  }
}

#include <SPI.h>
#include <MFRC522.h>
const int RST_PIN = 9;     // Reset pin
const int SS_PIN  = 10;    // SPI Slave Select pin
const int LED_PIN = 8;     // LED connected to digital pin 8
MFRC522 rfid(SS_PIN, RST_PIN);
bool ledState = false;
byte knownUID[] = { 0xCF, 0x5F, 0xB4, 0x79 };

void setup() {
  Serial.begin(9600);
  SPI.begin();              // Init SPI bus
  rfid.PCD_Init();          // Init MFRC522
  pinMode(LED_PIN, OUTPUT); // Set LED pin
  digitalWrite(LED_PIN, LOW);

  Serial.println("RFID reader initialized.");
}

void loop() {
  // Look for a new card
  if (!rfid.PICC_IsNewCardPresent() || !rfid.PICC_ReadCardSerial()) {
    delay(50);
    return;
  }
  // Print scanned UID
  Serial.print("UID tag: ");
  for (uint8_t i = 0; i < rfid.uid.size; i++) {
    Serial.print(rfid.uid.uidByte[i] < 0x10 ? " 0" : " ");
    Serial.print(rfid.uid.uidByte[i], HEX);
  }
  Serial.println();

  // Check if UID matches known tag
  if (isSameUID(rfid.uid.uidByte, knownUID, rfid.uid.size)) {
    ledState = !ledState;  // Toggle LED state
    digitalWrite(LED_PIN, ledState ? HIGH : LOW);
    Serial.println(ledState ? "LED ON" : "LED OFF");
    delay(1000); // prevent bouncing
  } else {
    Serial.println("Unknown tag.");
  }

  // Halt PICC
  rfid.PICC_HaltA();
}

bool isSameUID(byte *uid1, byte *uid2, byte length) {
  for (byte i = 0; i < length; i++) {
    if (uid1[i] != uid2[i]) return false;
  }
  return true;
}
